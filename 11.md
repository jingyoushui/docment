    public boolean helper(int i,int j,char[] str,int k){
        if(i<0||j<0||i>=rows||j>=cols){
            return false;
        }
        
        if(dp[i][j]==1||arr[i][j]!=str[k]){
            return false;
        }
        if(k==n-1){
           return true;
        }
        dp[i][j] = 1;
        for(int m=0;m<4;m++){
            if(helper(i+dir[m],j+dir[m+1],str,k+1)){
                return true;
            }
        }
        dp[i][j] = 0;
        return false;
        
    }
